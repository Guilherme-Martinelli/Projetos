#include <stdio.h>
#include <string.h>
#include <locale.h>

int main() {
		setlocale(LC_ALL, "Portuguese");

    char nome[50];

    printf("Digite um nome: ");
    fgets(nome, 50, stdin);

    int tamanho = strlen(nome);
    if (tamanho < 4) {
        printf("O nome tem menos de 4 letras.\n");
    } else {
        printf("As 4 primeiras letras do nome s„o: ");
        for (int i = 0; i < 4; i++) {
            printf("%c", nome[i]);
        }
        printf("\n");
    }

    return 0;
}
