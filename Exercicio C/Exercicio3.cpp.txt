#include <stdio.h>
#include <stdlib.h>
#include <locale.h>

int main() {
	setlocale(LC_ALL, "Portuguese");
    int numeros[5];  // Array para armazenar os 5 n·meros
    int menor, soma = 0, entre5e10 = 0;
    float media;

    // Ler os 5 n·meros fornecidos pelo usußrio
    for (int i = 0; i < 5; i++) {
        printf("Digite o %d║ n·mero: ", i + 1);
        scanf("%d", &numeros[i]);

        // Verificar se o n·mero estß entre 5 e 10
        if (numeros[i] >= 5 && numeros[i] <= 10) {
            entre5e10++;
        }

        // Calcular a soma dos n·meros
        soma += numeros[i];

        // Encontrar o menor n·mero
        if (i == 0) {
            menor = numeros[i];  // Inicializa com o primeiro n·mero
        } else if (numeros[i] < menor) {
            menor = numeros[i];
        }
    }

    // Calcular a mķdia
    media = (float)soma / 5;

    // Exibir os resultados
    printf("Menor n·mero digitado: %d\n", menor);
}
