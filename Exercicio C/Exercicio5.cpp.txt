#include <stdio.h>
#include <stdlib.h>
#include <locale.h>

int main() {
	setlocale(LC_ALL, "Portuguese");
    int quantidade, idade;
    int soma = 0; 
    float media;

    printf("Digite a quantidade de idades que deseja inserir: ");
    scanf("%d", &quantidade);

    for (int i = 0; i < quantidade; i++) {
        printf("Digite a idade %d: ", i + 1);
        scanf("%d", &idade);
        soma += idade; 
    }

    if (quantidade > 0) {
        media = (float)soma / quantidade;
        printf("A média das idades é: %.2f\n", media);
    } else {
        printf("Nenhuma idade foi inserida.\n");
    }

    return 0;
}
